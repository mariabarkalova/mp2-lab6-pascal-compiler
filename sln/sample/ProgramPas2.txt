program Example;
var num1, num2: integer; Res, d: double;
begin
num1:= 5;
num2:= 5;
Res := 0.0;
d := 0.0;
writeln('Введите четное целое число: ');
readln(num2);
writeln('Введите вещественное число: ');
readln(d);
if (num2 mod 2 = 0) then 
begin
Res := (num1 - num2 * 5 div 2) / (d * 2);
writeln('Результат = ', Res);
end
else
begin
writeln('Неверный ввод');
end;
end.
